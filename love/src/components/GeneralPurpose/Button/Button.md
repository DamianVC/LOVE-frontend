### Examples

Button with default props:

```jsx
<Button>Default</Button>
```

Different combinations of status and size:

#### Large size

```jsx
<div
  style={{
    display: 'flex',
    flexWrap: 'wrap',
    justifyContent: 'flext-start',
  }}
>
  <Button size="large" status="default">
    Default
  </Button>
  <Button size="large" status="primary">
    Primary
  </Button>
  <Button size="large" status="success">
    Success
  </Button>
  <Button size="large" status="info">
    Info
  </Button>
  <Button size="large" status="warning">
    Warning
  </Button>
  <Button size="large" status="danger">
    Danger
  </Button>
  <Button size="large" status="link">
    Link
  </Button>
</div>
```
